16) Demonstrate the Primary key, Unique key, Not Null key and Foreign key constraints using MySQL.

create table customer
(
	C_ID int,
    name varchar(20),
  	PRIMARY KEY (C_ID)
);

CREATE TABLE Orders
(
	O_ID int NOT NULL,
	Name varchar(20) NOT NULL unique,
	C_ID int,
	PRIMARY KEY (O_ID),
	FOREIGN KEY (C_ID) REFERENCES Customer(C_ID)
);
